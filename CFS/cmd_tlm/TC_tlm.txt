  APPEND_ITEM CCSDS_SUBSECS 16 UINT "CCSDS Telemetry Secondary Header (subseconds)"

  APPEND_ITEM COMMAND_ERROR_COUNTER 8 UINT "Command Error Counter" BIG_ENDIAN
